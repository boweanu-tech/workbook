\MajorQuestion{次の等式を証明しなさい。}
\BeginTwoCols
\QQ{$(a+b)^2-(a-b)^2=4ab$}{左辺 $=a^2+2ab+b^2-(a^2-2ab+b^2)=4ab$}
\QQ{$(x+y+z)^2-(x-y-z)^2=4x(y+z)$}{左辺 $=\{(x+y+z)+(x-y-z)\}\{(x+y+z)-(x-y-z)\}=4x(y+z)$}
\QQ{$a^3+b^3=(a+b)(a^2-ab+b^2)$}{右辺 $=a^3-a^2b+ab^2+a^2b-ab^2+b^3=a^3+b^3$}
\QQ{$\dfrac{1}{a} - \dfrac{1}{b}=\dfrac{b-a}{ab}$}{左辺 $=\dfrac{b-a}{ab}$}
\QQ{$\dfrac{x}{x+1}+\dfrac{1}{x+1}=1$}{左辺 $=\dfrac{x+1}{x+1}=1$}
\QQ{$\dfrac{a}{b}+\dfrac{b}{a}-2=\dfrac{(a-b)^2}{ab}$}{左辺 $=\dfrac{a^2+b^2-2ab}{ab}=\dfrac{(a-b)^2}{ab}$}
\EndTwoCols
