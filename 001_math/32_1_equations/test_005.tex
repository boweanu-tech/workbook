\Instruction{次の等式が $x$ についての恒等式となるように、定数 $a,b,c$ の値を求めなさい。}
\BeginTwoCols
\QQRow{$(a+1)x^2+(b-2)x+3\equiv 4x^2+5x+c$}{$a=3,\ b=7,\ c=3$}
{$ax^2+bx+c\equiv (2x-1)(x+3)$}{$a=2,\ b=5,\ c=-3$}
\QQRow{$(a-1)x^2+(b+2)x+c\equiv 3x^2-x+4$}{$a=4,\ b=-3,\ c=4$}
{$ax^3+bx^2+cx+2\equiv (x+1)(2x^2-x+2)$}{$a=2,\ b=1,\ c=1$}
\EndTwoCols

\Instruction{次の等式が $x$ についての恒等式となるように、定数 $a,b$ の値を求めなさい。}
\BeginTwoCols
\QQRow{$\dfrac{1}{(x-1)(x+2)}\equiv\dfrac{a}{x-1}+\dfrac{b}{x+2}$}{$a=\dfrac{1}{3},\ b=-\dfrac{1}{3}$}
{$\dfrac{2x+1}{(x+1)(x-2)}\equiv\dfrac{a}{x+1}+\dfrac{b}{x-2}$}{$a=\dfrac{1}{3},\ b=\dfrac{5}{3}$}
\EndTwoCols

\Instruction{次の等式がすべての $x$ について成り立つとき、定数 $a,b$ の値を求めなさい。}
\BeginTwoCols
\QQRow{$x^2+ax+b\equiv (x-2)(x+3)+4$}{$a=1,\ b=-2$}
{$(a+b)x+(2a-b)\equiv 5x+1$}{$a=2,\ b=3$}
\EndTwoCols
