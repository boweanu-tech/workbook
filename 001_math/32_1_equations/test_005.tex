\Instruction{次の等式を証明しなさい。}
\BeginTwoCols
\QQ{$\dfrac{a}{b}+\dfrac{b}{a}-2=\dfrac{(a-b)^2}{ab}$}{左辺 $=\dfrac{a^2+b^2-2ab}{ab}=\dfrac{(a-b)^2}{ab}$}
\QQ{$\dfrac{a+b}{a}+\dfrac{a+b}{b}=\dfrac{(a+b)^2}{ab}$}{左辺 $=\dfrac{b(a+b)+a(a+b)}{ab}=\dfrac{(a+b)^2}{ab}$}
\EndTwoCols

\Instruction{$a+b+c=0$ のとき、次の等式を証明しなさい。}
\BeginTwoCols
\QQ{$a^3+b^3+c^3=3abc$}{$a+b=-c$ より、$a^3+b^3+c^3=(a+b)^3-3ab(a+b)+c^3=-c^3+3abc+c^3=3abc$}
\QQ{$a^2+b^2+c^2=-2(ab+bc+ca)$}{$(a+b+c)^2=0$ より、$a^2+b^2+c^2+2ab+2bc+2ca=0$}
\EndTwoCols
