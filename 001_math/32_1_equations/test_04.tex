\Instruction{次の等式が恒等式となるように、定数を求めなさい。}
\BeginTwoCols
\QQ{$(a+b)x^2+(a-b)x+3=5x^2+x+c$}{$a=3,\ b=2,\ c=3$}
\QQ{$ax^2+bx+c=(2x-1)(x+3)$}{$a=2,\ b=5,\ c=-3$}
\QQ{$\dfrac{1}{(x-1)(x+2)}=\dfrac{a}{x-1}+\dfrac{b}{x+2}$}{$a=\dfrac{1}{3},\ b=-\dfrac{1}{3}$}
\EndTwoCols

\Instruction{次の等式がすべての $x,y$ について成り立つとき、定数 $a,b$ の値を求めなさい。}
\BeginTwoCols
\QQ{$a(x+y)+b(x-y)=5x+y$}{$a=3,\ b=2$}
\QQ{$a(x^2+2xy)+b(y^2-xy)=3x^2+4xy+2y^2$}{$a=3,\ b=2$}
\EndTwoCols
