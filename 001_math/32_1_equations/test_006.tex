\Instruction{次の不等式を証明しなさい。}
\BeginTwoCols
\QQ{$x^2-4x+7>0\quad (x\text{ は実数})$}{左辺 $=(x-2)^2+3>0$}
\QQ{$a^2+b^2+c^2\geqq ab+bc+ca$}{左辺$-$右辺 $=\dfrac{1}{2}\{(a-b)^2+(b-c)^2+(c-a)^2\}\geqq 0$}
\QQ{$x^2+2xy+2y^2-4x-6y+5\geqq 0$}{左辺 $=(x+y-2)^2+(y-1)^2\geqq 0$}
\EndTwoCols

\Instruction{相加平均と相乗平均の関係を用いて、次の不等式を証明し、等号が成り立つときの文字の値を求めなさい。}
\BeginTwoCols
\QQ{$a+\dfrac{1}{a+3}\geqq -1\quad (a>-3)$}{$a+\dfrac{1}{a+3}=(a+3)+\dfrac{1}{a+3}-3\geqq 2\sqrt{(a+3)\cdot\dfrac{1}{a+3}}-3=-1$。等号は $a+3=\dfrac{1}{a+3}$、すなわち $a=-2$ のとき。}
\QQ{$x+\dfrac{9}{x}\geqq 6\quad (x>0)$}{$x+\dfrac{9}{x}\geqq 2\sqrt{x\cdot\dfrac{9}{x}}=6$。等号は $x=\dfrac{9}{x}$、すなわち $x=3$ のとき。}
\QQ{$a+\dfrac{4}{a-1}\geqq 5\quad (a>1)$}{$a+\dfrac{4}{a-1}=(a-1)+\dfrac{4}{a-1}+1\geqq 2\sqrt{4}+1=5$。等号は $a-1=\dfrac{4}{a-1}$、すなわち $a=3$ のとき。}
\EndTwoCols
