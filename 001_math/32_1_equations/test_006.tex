\MajorQuestion{次の問いに答えなさい。}

\Instruction{次の不等式を証明しなさい。}
\BeginTwoCols
\QQ{$a^2+b^2\geqq 2ab$}{左辺$-$右辺 $=(a-b)^2\geqq 0$}
\QQ{$x^2-4x+7>0$}{左辺 $=(x-2)^2+3>0$}
\QQ{$a^2+b^2+c^2\geqq ab+bc+ca$}{左辺$-$右辺 $=\dfrac{1}{2}\{(a-b)^2+(b-c)^2+(c-a)^2\}\geqq 0$}
\EndTwoCols

\Instruction{正の数 $a,b$ について、相加平均と相乗平均の関係を用いて、次の不等式を証明しなさい。}
\BeginTwoCols
\QQ{$a+b\geqq 2\sqrt{ab}$}{相加平均と相乗平均の関係より、$\dfrac{a+b}{2}\geqq\sqrt{ab}$}
\QQ{$x+\dfrac{4}{x}\geqq 4\quad (x>0)$}{相加平均と相乗平均の関係より、$x+\dfrac{4}{x}\geqq 2\sqrt{x\cdot\dfrac{4}{x}}=4$}
\QQ{$a+b+c\geqq 3\sqrt[3]{abc}$}{相加平均と相乗平均の関係より、$\dfrac{a+b+c}{3}\geqq\sqrt[3]{abc}$}
\EndTwoCols
