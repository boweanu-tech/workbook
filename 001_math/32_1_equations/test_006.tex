\MajorQuestion{次の問いに答えなさい。}

\Instruction{次の不等式を証明しなさい。等号があるものは、等号成立条件も答えなさい。ここで$x,y$は実数、$a,b,c$は正の実数とする。}
\BeginTwoCols
\QQ{$x^2-4x+7>0\quad $}{左辺 $=(x-2)^2+3>0$}
\QQ{$a+\dfrac{25}{a+3}\geqq 7$}{$a+\dfrac{25}{a+3}=(a+3)+\dfrac{25}{a+3}-3\geqq 2\sqrt{(a+3)\cdot\dfrac{25}{a+3}}-3=7$。等号は $a+3=\dfrac{25}{a+3}$、すなわち $a=2$ のとき。}
\QQ{$x^2+2xy+2y^2-4x-6y+5\geqq 0$}{左辺 $=(x+y-2)^2+(y-1)^2\geqq 0$。等号は $x=1,\ y=1$ のとき。}
\QQ{$\sqrt{a^2+b^2}\geqq \dfrac{a+b}{\sqrt{2}}$}{両辺は正である。両辺を2乗して差をとると、$a^2+b^2-\dfrac{(a+b)^2}{2}=\dfrac{(a-b)^2}{2}\geqq 0$。等号は $a=b$ のとき。}
\QQ{$|x|+|y|\geqq |x+y|$}{両辺は0以上である。両辺を2乗して差をとると、$(|x|+|y|)^2-|x+y|^2=2(|xy|-xy)\geqq 0$。等号は $xy\geqq 0$ のとき。}
\EndTwoCols
