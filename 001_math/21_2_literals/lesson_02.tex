\ifLMAnswerMode\else\hfill{\small 02\_数量を文字式で表す}\par\vspace{2mm}\fi

\Instruction{個数や代金を、文字式で表しなさい。}
\QQ{1個$a$円のりんごを$6$個買ったときの代金}{$6a$円}
\QQ{1冊$x$円のノートを$4$冊買ったときの代金}{$4x$円}
\QQ{$1$本$b$円の鉛筆を$8$本買ったときの代金}{$8b$円}
\QQ{$1$個$y$円の消しゴムを$3$個と、$1$冊$120$円のノートを$2$冊買ったときの代金}{$3y+240$円}
\QQ{$1$個$a$円のみかんを$5$個と、$1$個$b$円のりんごを$3$個買ったときの代金}{$5a+3b$円}

\Instruction{残りやおつりを、文字式で表しなさい。}
\QQ{$x$円を持っていて、$150$円のノートを$2$冊買ったときの残金}{$x-300$円}
\QQ{$1000$円を出して、$1$個$a$円の品物を$4$個買ったときのおつり}{$1000-4a$円}
\QQ{$a$個のあめを$1$人に$3$個ずつ、$b$人に配ったときの残りの個数}{$a-3b$個}
\QQ{$x$個のボールから、$1$箱に$12$個ずつ$n$箱分を取り出したときの残りの個数}{$x-12n$個}

\Instruction{単位に注意して、文字式で表しなさい。}
\QQ{$a$ mをcmで表した長さ}{$100a$ cm}
\QQ{$x$ kmをmで表した長さ}{$1000x$ m}
\QQ{$b$ cmをmで表した長さ}{$\frac{b}{100}$ m}
\QQ{$y$ gをkgで表した重さ}{$\frac{y}{1000}$ kg}
\QQ{$t$分を秒で表した時間}{$60t$秒}
\QQ{$a$ LをmLで表した体積}{$1000a$ mL}

\ifLMAnswerMode\end{minipage}\columnbreak\noindent\begin{minipage}{\linewidth}\else\clearpage\hfill{\small 02\_数量を文字式で表す}\par\vspace{2mm}\fi

\Instruction{割合に注意して、文字式で表しなさい。}
\QQ{$x$円の$10\%$}{$\frac{x}{10}$円}
\QQ{$a$円の$25\%$}{$\frac{a}{4}$円}
\QQ{$b$人の$40\%$}{$\frac{2b}{5}$人}
\QQ{$x$円の$3$割}{$\frac{3x}{10}$円}
\QQ{$x$円の商品を$20\%$引きで買うときの代金}{$\frac{4x}{5}$円}
\QQ{$a$円の商品に$8\%$の消費税がついたときの税込価格}{$\frac{27a}{25}$円}

\Instruction{速さ・時間・道のりを、文字式で表しなさい。}
\QQ{分速$x$ mで$7$分歩いたときの道のり}{$7x$ m}
\QQ{時速$a$ kmで$3$時間進んだときの道のり}{$3a$ km}
\QQ{$x$ kmを$2$時間で進んだときの速さ}{時速$\frac{x}{2}$ km}

\Instruction{図形や数に関する数量を、文字式で表しなさい。}
\QQ{縦$a$ cm、横$7$ cmの長方形の面積}{$7a$ cm$^2$}
\QQ{縦$x$ cm、横$5$ cmの長方形の周の長さ}{$2x+10$ cm}
\QQ{底辺が$a$ cm、高さが$h$ cmの三角形の面積}{$\frac{ah}{2}$ cm$^2$}
\QQ{$x$の$4$倍と$7$の和}{$4x+7$}
\QQ{$n$を整数とするとき、偶数を表す式}{$2n$}
\QQ{$n$を整数とするとき、奇数を表す式}{$2n+1$}
