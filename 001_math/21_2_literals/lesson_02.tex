\Instruction{次の数量を、文字式で表しなさい。}
\QQ{1個$a$円のりんごを$6$個買ったときの代金}{$6a$円}
\QQ{1冊$x$円のノートを$4$冊買ったときの代金}{$4x$円}
\QQ{$1$本$b$円の鉛筆を$8$本買ったときの代金}{$8b$円}
\QQ{$1$個$y$円の消しゴムを$3$個と、$1$冊$120$円のノートを$2$冊買ったときの代金}{$3y+240$円}
\QQ{$1$個$a$円のみかんを$5$個と、$1$個$b$円のりんごを$3$個買ったときの代金}{$5a+3b$円}
\QQ{$1$人に$x$枚ずつプリントを配る。$7$人に配るときに必要なプリントの枚数}{$7x$枚}
\QQ{$n$人に$1$人$4$枚ずつカードを配り、さらに予備を$10$枚用意するときのカードの枚数}{$4n+10$枚}
\QQ{$a$個のあめを$1$人に$3$個ずつ、$b$人に配ったときの残りの個数}{$a-3b$個}

\Instruction{次の「残り」や「おつり」を、文字式で表しなさい。}
\QQ{$x$円を持っていて、$150$円のノートを$2$冊買ったときの残金}{$x-300$円}
\QQ{$1000$円を出して、$1$個$a$円の品物を$4$個買ったときのおつり}{$1000-4a$円}
\QQ{$y$円を持っていて、$1$本$80$円のペンを$a$本買ったときの残金}{$y-80a$円}
\QQ{$x$個のボールから、$1$箱に$12$個ずつ$a$箱分を取り出したときの残りの個数}{$x-12a$個}
\QQ{$a$円の商品を$300$円引きで買うときの代金}{$a-300$円}
\QQ{$1$個$x$円の商品を$5$個買い、袋代$10$円を足したときの代金}{$5x+10$円}
\QQ{$1$枚$a$円のコピーを$b$枚とり、$500$円を出したときのおつり}{$500-ab$円}

\Instruction{単位に注意して、次の数量を文字式で表しなさい。}
\QQ{$a$ mをcmで表した長さ}{$100a$ cm}
\QQ{$x$ kmをmで表した長さ}{$1000x$ m}
\QQ{$b$ cmをmで表した長さ}{$\frac{b}{100}$ m}
\QQ{$y$ gをkgで表した重さ}{$\frac{y}{1000}$ kg}
\QQ{$t$分を秒で表した時間}{$60t$秒}
\QQ{$x$時間を分で表した時間}{$60x$分}
\QQ{$a$ LをmLで表した体積}{$1000a$ mL}
\QQ{$b$ mLをLで表した体積}{$\frac{b}{1000}$ L}
\QQ{分速$x$ mで$7$分歩いたときの道のり}{$7x$ m}
\QQ{時速$a$ kmで$3$時間進んだときの道のり}{$3a$ km}
\QQ{$x$ kmを$2$時間で進んだときの速さ}{時速$\frac{x}{2}$ km}
\QQ{$y$ mを$5$分で進んだときの速さ}{分速$\frac{y}{5}$ m}
