\Instruction{次の方程式のうち、二次方程式であるものをすべて選びなさい。}

\QQRow{$2x^2=8$}{二次方程式}{$x^2+4=2x^2$}{二次方程式}
\QQRow{$(x+1)(x-3)=0$}{二次方程式}{$\dfrac{1}{x}+2=0$}{二次方程式ではない}
\EndTwoCols

\Instruction{次の値が、与えられた二次方程式の解であるかを答えなさい。}
\BeginTwoCols
\QQRow{$x^2-7x+12=0$ において $x=3$}{$3^2-7\cdot3+12=0$ より、解である。}{$x^2-7x+12=0$ において $x=5$}{$5^2-7\cdot5+12=2\neq0$ より、解ではない。}
\QQRow{$2x^2-5x-3=0$ において $x=3$}{$2\cdot3^2-5\cdot3-3=0$ より、解である。}{$2x^2-5x-3=0$ において $x=-\dfrac{1}{2}$}{$2\left(-\dfrac{1}{2}\right)^2-5\left(-\dfrac{1}{2}\right)-3=0$ より、解である。}
\QQRow{$x^2+2x-8=0$ において $x=-4$}{$(-4)^2+2\cdot(-4)-8=0$ より、解である。}{$x^2+2x-8=0$ において $x=4$}{$4^2+2\cdot4-8=16\neq0$ より、解ではない。}
\EndTwoCols
