\MajorQuestionLesson{二次方程式の意味と解}
\SetRowSpace{5mm}

\TypeQuestionSingle{次の方程式が二次方程式である場合は○、そうでない場合は×を答えなさい。}
\QQRow{$x^2-5x+6=0$}{}{$3x+7=0$}{}
\QQRow{$2x^2=8$}{}{$5x-1=0$}{}
\QQRow{$x^2+4x=0$}{}{$x+2=9$}{}
\QQRow{$4x^2-1=0$}{}{$\dfrac{1}{x}+2=0$}{}

\TypeQuestionDouble{次の方程式が二次方程式である場合は○、そうでない場合は×を答えなさい。ただし、必要なら左辺を整理して考えなさい。}
\QQRow{$x^2+4=2x^2$}{}{$x^2+3x=x^2-1$}{}
\QQRow{$(x+1)(x-3)=0$}{}{$(x+2)^2=x^2+5$}{}
\QQRow{$2x^2-3x=4x^2+1$}{}{$(x-1)(x+1)=x^2$}{}
\QQRow{$3(x+1)=x^2-2$}{}{$x(x-4)=2x$}{}

\TypeQuestionSingle{次の方程式を、すべての項を左辺に移して整理しなさい。}
\QQRow{$x^2=5x-6$}{}{$2x^2+3=7x$}{}
\QQRow{$x^2+4x=12$}{}{$3x^2=2x+8$}{}
\QQRow{$(x+1)(x-2)=0$}{}{$(x-3)^2=4$}{}
\QQRow{$2(x+1)^2=x+5$}{}{$x(x-5)=3x-2$}{}
