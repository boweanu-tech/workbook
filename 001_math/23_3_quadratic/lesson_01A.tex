\MajorQuestionLesson{二次方程式の意味と解}
\SetRowSpace{5mm}

\TypeQuestionSingle{次の二次方程式について、$x^2$ の係数、$x$ の係数、定数項を順に答えなさい。}{keisu}
\QQRow{$6-5x+x^2=0$}{$1,\ -5,\ 6$}{$3x-1+2x^2=0$}{$2,\ 3,\ -1$}
\QQRow{$7-x^2+4x=0$}{$-1,\ 4,\ 7$}{$-9+5x^2=0$}{$5,\ 0,\ -9$}
\QQRow{$x+3x^2=0$}{$3,\ 1,\ 0$}{$8+x^2=0$}{$1,\ 0,\ 8$}
\QQRow{$-6x+4x^2=0$}{$4,\ -6,\ 0$}{$2x^2-7x=0$}{$2,\ -7,\ 0$}

\TypeQuestionSingle{次の方程式が二次方程式である場合は○、そうでない場合は×を答えなさい。}{niji}
\QQRow{$x^2-5x+6=0$}{○}{$3x+7=0$}{×}
\QQRow{$2x^2=8$}{○}{$5x-1=0$}{×}
\QQRow{$x+2=9$}{×}{$x^2+4x=0$}{○}
\QQRow{$4x^2-1=0$}{○}{$\dfrac{1}{x}+2=0$}{×}

\TypeQuestionDouble{次の方程式が二次方程式である場合は○、そうでない場合は×を答えなさい。ただし、必要なら両辺を整理して考えなさい。}{seiri}
\QQRow{$x^2+4=2x^2$}{○}{$x(x-4)=2x$}{○}
\QQRow{$(x+2)^2=x^2+5$}{×}{$(x+1)(x-3)=0$}{○}
\QQRow{$2x^2-3x=4x^2+1$}{○}{$(x-1)(x+1)=x^2$}{×}
\QQRow{$3(x+1)=x^2-2$}{○}{$x^2+3x=x^2-1$}{×}

\LessonPageBreak

\TypeQuestionSingle{次の数が、与えられた二次方程式の解である場合は○、そうでない場合は×を答えなさい。}{kaihan}
\QQRow{$x=2$ は $x^2-5x+6=0$ の解である}{○}{$x=1$ は $x^2-5x+6=0$ の解である}{×}
\QQRow{$x=-3$ は $x^2-9=0$ の解である}{○}{$x=0$ は $x^2-9=0$ の解である}{×}
\QQRow{$x=-4$ は $x^2+2x-8=0$ の解である}{○}{$x=4$ は $x^2+2x-8=0$ の解である}{×}
\QQRow{$x=3$ は $2x^2-5x-3=0$ の解である}{○}{$x=1$ は $2x^2-5x-3=0$ の解である}{×}

\TypeQuestionDouble{次の候補のうち、与えられた二次方程式の解であるものをすべて答えなさい。なければ無しと答えなさい。}{kouho}
\QQ{$-4,\ -2,\ 0,\ 2,\ 4$ のうち、$x^2+2x-8=0$ の解であるもの}{$-4,\ 2$}
\QQ{$-1,\ 2,\ 3$ のうち、$x^2-5x-6=0$ の解であるもの}{$-1$}
\QQ{$-3,\ -1,\ 1,\ 3$ のうち、$x^2-9=0$ の解であるもの}{$-3,\ 3$}
\QQ{$-2,\ -\dfrac{1}{2},\ \dfrac{1}{2},\ 3$ のうち、$2x^2-5x-3=0$ の解であるもの}{$-\dfrac{1}{2},\ 3$}
\QQ{$-4,\ -2,\ 1,\ 4$ のうち、$x^2-5x+6=0$ の解であるもの}{無し}

\TypeQuestionSingle{指定された数が解になるように、$a$ の値を求めなさい。}{gyaku}
\QQ{$x=1$ が $x^2-ax-6=0$ の解である}{$a=-5$}
\QQ{$x=2$ が $ax^2-3x-2=0$ の解である}{$a=2$}
\QQ{$x=3$ が $x^2-6x+3a=0$ の解である}{$a=3$}
\QQ{$x=-3$ が $2x^2+(-a+1)x-3=0$ の解である}{$a=-4$}
