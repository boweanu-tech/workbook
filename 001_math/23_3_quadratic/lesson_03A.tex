\MajorQuestionLesson{因数分解と解の公式}
\SetRowSpace{8mm}

\TypeQuestionSingle{次の方程式を解きなさい。}
%大問を３つに分ける
%「項が２つパターン１」x^2-9=0とか　4題
%「項が２つパターン２」x^2-6x=0とか　4題
%「項が３つパターン」10題
%の３つ 
\QQRow{$x^2-5x+6=0$}{$x=2,\ 3$}{$x^2+x-12=0$}{$x=3,\ -4$}
\QQRow{$x^2-9=0$}{$x=\pm3$}{$x^2+4x=0$}{$x=0,\ -4$}
\QQRow{$x^2-7x+12=0$}{$x=3,\ 4$}{$x^2+2x-15=0$}{$x=3,\ -5$}

\LessonPageBreak

\TypeQuestionSingle{次の方程式を、解の公式を使って解きなさい。}
%x^2の係数が１、x１次の係数が奇数のものに統一
\QQRow{$2x^2+3x-4=0$}{$x=\dfrac{-3\pm\sqrt{41}}{4}$}{$3x^2-5x-1=0$}{$x=\dfrac{5\pm\sqrt{37}}{6}$}
\QQRow{$2x^2-4x-7=0$}{$x=1\pm\dfrac{3\sqrt{2}}{2}$}{$5x^2+2x-1=0$}{$x=\dfrac{-1\pm\sqrt{6}}{5}$}
\QQRow{$3x^2+2x-6=0$}{$x=\dfrac{-1\pm\sqrt{19}}{3}$}{$7x^2-4x-2=0$}{$x=\dfrac{2\pm3\sqrt{2}}{7}$}

\TypeQuestionSingle{次の方程式を、解の公式か、又は平方完成を使って解きなさい。}
%x１次の係数が偶数のものに統一
%x^2の係数は１以外で統一
\QQRow{$x^2-2x-1=0$}{$x=1\pm\sqrt{2}$}{$x^2+6x+2=0$}{$x=-3\pm\sqrt{7}$}
\QQRow{$x^2-6x+1=0$}{$x=3\pm2\sqrt{2}$}{$x^2+4x-3=0$}{$x=-2\pm\sqrt{7}$}
